# Python Client

Official Python SDK for the Nebula Memory API.

## Installation

```bash
pip install nebula-client
```

## Setup

```bash
# Set environment variable
export NEBULA_API_KEY="your_api_key_here"
```

```python
from nebula_client import NebulaClient

# Automatically uses NEBULA_API_KEY environment variable
client = NebulaClient()
```

## Basic Usage

```python
from nebula_client import NebulaClient

client = NebulaClient()

# Create a collection
collection = client.create_collection(name="my_notes", description="Personal notes")

# Store a memory
memory_id = client.store_memory({
    "collection_id": collection.id,
    "content": "Python is great for data science",
    "metadata": {"topic": "programming", "language": "python"}
})

# Search memories
results = client.search(
    query="data science",
    collection_ids=[collection.id],
    limit=10
)

for result in results:
    print(f"Score: {result.score:.2f}")
    print(f"Content: {result.content}")
```

## Async Support

The Python SDK includes a dedicated async client:

```python
import asyncio
from nebula_client import AsyncNebulaClient

async def main():
    # Create async client (uses NEBULA_API_KEY env var)
    client = AsyncNebulaClient()

    # All operations are async
    collection = await client.create_collection(name="async_notes", description="Async example")

    memory_id = await client.store_memory({
        "collection_id": collection.id,
        "content": "Async operations are fast",
        "metadata": {"type": "example"}
    })

    results = await client.search(
        query="async",
        collection_ids=[collection.id]
    )

    for result in results:
        print(f"{result.content}")

    # Clean up
    await client.aclose()

asyncio.run(main())
```

### Async Context Manager

```python
import asyncio
from nebula_client import AsyncNebulaClient

async def main():
    # Automatically handles cleanup
    async with AsyncNebulaClient() as client:
        collection = await client.create_collection(name="notes", description="My notes")

        memory_id = await client.store_memory({
            "collection_id": collection.id,
            "content": "Context managers are clean"
        })

        results = await client.search(
            query="clean",
            collection_ids=[collection.id]
        )

        print(f"Found {len(results)} results")

asyncio.run(main())
```
